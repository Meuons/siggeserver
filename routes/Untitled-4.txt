CREATE TABLE categories
              (
                category_id INTEGER PRIMARY KEY,
				title TEXT NOT NULL,
                description TEXT NOT NULL,
				icon TEXT NOT NULL, 
				account INT NOT NULL,
                wage_type INT NOT NULL,
                deductible BIT DEFAULT 0,
                admin_only BIT DEFAULT 0, "tax" INTEGER NOT NULL,
                 FOREIGN KEY (account) REFERENCES account(account_id)
                 FOREIGN KEY (wage_type) REFERENCES wage_types(type_id)
              )
